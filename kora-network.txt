# FILE: ./audio-processing-service/src/main.py
import os
import shutil

from fastapi import FastAPI, UploadFile, File, HTTPException, status

from processor import AudioProcessor
from audio_utils import save_upload_to_temp, convert_to_wav, load_audio

app = FastAPI(title="Audio Processing Service")

processor = AudioProcessor()


def cleanup(path: str | None):
    if path and os.path.exists(path):
        os.remove(path)


@app.get("/health")
def health():
    return {
        "status": "ok",
        "service": "audio-processing-service",
    }


@app.post("/process")
async def process(file: UploadFile = File(...)):
    if not file.filename:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail="file is required",
        )

    input_path = None
    wav_path = None

    try:
        input_path = save_upload_to_temp(await file.read())

        wav_path = convert_to_wav(input_path)

        audio, sr = load_audio(wav_path)

        result = processor.process(audio)

        return result

    except Exception as e:
        raise HTTPException(
            status_code=500,
            detail=str(e),
        )

    finally:
        cleanup(input_path)
        cleanup(wav_path)
# FILE: ./audio-processing-service/src/audio_utils.py
import tempfile
import subprocess
import numpy as np
import soundfile as sf


def save_upload_to_temp(file_bytes: bytes, suffix=".webm") -> str:
    tmp = tempfile.NamedTemporaryFile(delete=False, suffix=suffix)
    tmp.write(file_bytes)
    tmp.close()
    return tmp.name


def convert_to_wav(input_path: str) -> str:
    output_path = tempfile.NamedTemporaryFile(delete=False, suffix=".wav").name

    subprocess.run(
        [
            "ffmpeg",
            "-y",
            "-i",
            input_path,
            "-ar",
            "16000",
            "-ac",
            "1",
            output_path,
        ],
        check=True,
        capture_output=True,
    )

    return output_path


def load_audio(path: str):
    audio, sr = sf.read(path)
    return audio, sr
# FILE: ./audio-processing-service/src/processor.py
import numpy as np
import webrtcvad
import collections
import scipy.signal as signal


class AudioProcessor:
    def __init__(
        self,
        sample_rate: int = 16000,
        frame_ms: int = 30,
        vad_mode: int = 2,
        smoothing_window: int = 7,
        noise_reduce_strength: float = 0.7,
        min_speech_ratio: float = 0.6,
    ):
        self.sample_rate = sample_rate
        self.frame_ms = frame_ms
        self.vad = webrtcvad.Vad(vad_mode)

        self.frame_size = int(sample_rate * frame_ms / 1000)
        self.buffer = collections.deque(maxlen=smoothing_window)

        self.noise_profile = None
        self.noise_reduce_strength = noise_reduce_strength
        self.min_speech_ratio = min_speech_ratio

    # -------------------------
    def _frame_generator(self, audio: np.ndarray):
        for i in range(0, len(audio), self.frame_size):
            yield audio[i:i + self.frame_size]

    # -------------------------
    def _is_speech(self, frame: np.ndarray):
        if len(frame) < self.frame_size:
            return False

        frame_int16 = (frame * 32767).astype(np.int16).tobytes()
        return self.vad.is_speech(frame_int16, self.sample_rate)

    # -------------------------
    def _smooth(self, decision: bool):
        self.buffer.append(1 if decision else 0)
        return (sum(self.buffer) / len(self.buffer)) > self.min_speech_ratio

    # -------------------------
    def build_noise_profile(self, audio: np.ndarray):
        f, t, Zxx = signal.stft(audio, nperseg=256)
        self.noise_profile = np.mean(np.abs(Zxx), axis=1)

    # -------------------------
    def reduce_noise(self, audio: np.ndarray):
        if self.noise_profile is None:
            return audio

        f, t, Zxx = signal.stft(audio, nperseg=256)
        magnitude = np.abs(Zxx)
        phase = np.angle(Zxx)

        threshold = self.noise_profile[:, None] * self.noise_reduce_strength
        mask = magnitude > threshold

        cleaned = mask * magnitude

        _, recovered = signal.istft(cleaned * np.exp(1j * phase))
        return recovered.astype(np.float32)

    # -------------------------
    def process(self, audio: np.ndarray):
        audio = audio.astype(np.float32)
        audio = audio / (np.max(np.abs(audio)) + 1e-8)

        speech_mask = []
        speech_ratio_frames = []

        for frame in self._frame_generator(audio):
            if len(frame) < self.frame_size:
                continue

            raw = self._is_speech(frame)
            smooth = self._smooth(raw)

            speech_mask.extend([smooth] * len(frame))
            speech_ratio_frames.append(smooth)

        speech_ratio = sum(speech_ratio_frames) / max(len(speech_ratio_frames), 1)

        cleaned = self.reduce_noise(audio)

        return {
            "audio": cleaned,
            "speech_ratio": float(speech_ratio),
            "is_speech": speech_ratio > 0.2,
        }
